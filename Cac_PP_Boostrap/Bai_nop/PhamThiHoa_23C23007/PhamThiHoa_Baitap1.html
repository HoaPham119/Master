# nh ngha hm full_function: bao gm y  cc bc: 
# - (a) To ngu nhin mt mu vi c n = 200, t phn phi ny, hy tnh trung bnh v  lch chun mu, t y, suy ra phn phi mu cho trung bnh mu;
# - (b) p dng phng php bootstrap  xc nh phn phi mu ca mu va to, v so snh vi phn phi mu c xc nh bng l thuytNhp vo mean, sd v n

full_function <- function(mean, sd, n) {
    # To mu n gi tr vi trung bnh = mean v  lch chun = sd
    sample <- rnorm(n, mean = mean, sd = sd)

    # Xem 10 mu u tin
    cat("10 mau dau tien:\n", sample[1:10], "\n")

    # Tnh trung bnh v  lch chun ca mu t d liu va c to
    sample_mean <- mean(sample)
    sample_sd <- sd(sample)
    # In kt qu
    cat("Trung binh cua mau:", sample_mean, "\n")
    cat("Do lech chuan cua mau:", sample_sd, "\n")

    # Suy ra phn phi ca trung bnh mu
    # Vi trung bnh ca mu l sample_mean v  lch chun ca mu l sample_sd,
    # phn phi ca trung bnh mu l mt phn phi chun vi:
    # trung bnh = sample_mean 
    # v  lch chun = sample_sd/ sqrt(n)
    population_mean <- sample_mean
    population_sd <- sample_sd / sqrt(n)

    cat("Phan phoi cho trung binh mau: mean =", population_mean, ", sd =", population_sd, "\n")

    # V histogram ca mu
    hist(sample, main = "Histogram cua mau", xlab = "Gia tri", ylab = "Tan suat", col = "skyblue", border = "#EEF1F2")
    
    # S dng hm boot()  thc hin bootstrap
    boot_result <- boot(data = sample, statistic = boot_mu_fun, R = 1000)
    # Hin th kt qu boostrao
    print(boot_result)
    # V histogram ca cc trung bnh mu bootstrap
    hist(boot_result$t, main = "Phan phoi bootstrap cua trung binh mau", xlab = "Trung binh mau", ylab = "Tan suat", col = "skyblue", border = "#5EC0C5")

    # So snh phn phi mu bootstrap vi phn phi l thuyt ca trung bnh mu
    abline(v = population_mean, col = "red", lwd = 2)
}

# Hm tnh ton theta
tinh_theta_mu <- function(data, indices) {
  d <- data[indices, ]  # ly mu bootstrap
  # Tch thnh 2 nhm X l nhm 0 v Y l nhm 1
  X <- d$scores[d$group == 0] 
  Y <- d$scores[d$group == 1]
  
  n <- length(X)
  m <- length(Y)
  count <- 0
  
  for (x in X) {
    for (y in Y) {
      if (x < y) {
        count <- count + 1
      }
    }
  }
  
  theta_mu <- count / (n * m)
  return(theta_mu)
}
